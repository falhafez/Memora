/* =======================

$knowledge = [];

/* =======================

$context = "";

foreach($knowledge as $item){

        if(mb_stripos($message,$keyword)!==false){

            $context .=
            $item["title"] . "\n" .
            $item["content"] . "\n\n";

            break;
        }

    }

}


/* =======================
   ساخت Prompt
======================= */

if($context!=""){

/* =======================
